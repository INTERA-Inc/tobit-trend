{
    EVENT <- NULL
    X <- copy(X)
    X <- data.table(X)
    MIN <- min(X$EVENT)
    MAX <- max(X$EVENT)
    if (class(MIN)[1] == "Date") {
        DT <- data.table(EVENT = seq(MIN, MAX, by = "day"))
    }
    else if (class(MIN)[1] == "POSIXct") {
        DT <- data.table(EVENT = seq(MIN, MAX, by = "hour"))
    }
    setkey(DT, EVENT)
    setkey(X, EVENT)
    DT <- X[DT]
    DT$INTERP <- na.approx(DT$STAGE)
    DT$NAME <- unique(X$NAME)
    return(DT)
}
