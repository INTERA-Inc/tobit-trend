{
    LOG <- match.arg(LOG)
    if (LOG == "log") {
        FORM <- paste0("log(", LHS, ")~")
    }
    else if (LOG == "log10") {
        FORM <- paste0("log10(", LHS, ")~")
    }
    else {
        FORM <- paste0(LHS, "~")
    }
    FORM <- paste0(FORM, paste(RHS, collapse = "+"))
    return(as.formula(FORM))
}
