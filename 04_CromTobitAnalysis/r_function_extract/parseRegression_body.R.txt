### sspaTrendAnalysis::parseRegression

{
    LOG <- match.arg(LOG)
    TS <- match.arg(TS)
    if (!"TERM" %in% names(X)) {
        FORM <- createFormula(LHS = LHS, RHS = RHS, LOG)
        return(list(TS = TS, LOG = LOG, FORM = FORM, DATA = X))
    }
    else {
        FORM <- createFormula(LHS = LHS, RHS = RHS, LOG)
        TERMS <- sort(unique(X$TERM))
        return(list(TS = TS, LOG = LOG, FORM = FORM, TERMS = TERMS, 
            DATA = X))
    }
}
