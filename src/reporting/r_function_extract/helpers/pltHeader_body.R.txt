function(X){

            #---------------------------------------------------------------------------------#
            #--Extract Well Name and Distance to the River and Number of Data Points--#
            NAME <- X@NAME
            #---------------------------------------------------------------------------------#

            #---------------------------------------------------------------------------------#
            #--Additional Plot Label--#
            LABEL <- paste('Distance to River:',round(X@DATA$DIST[1],0),'m','\n',
                           'Number of Trends Calculated:',1)
            #---------------------------------------------------------------------------------#

            #---------------------------------------------------------------------------------#
            #--Plot Header--#
            plot.new()
            mtext(NAME,side=1, line=-4, cex=1.5, font=3)
            mtext(LABEL,side=1, line=0, cex=0.8, font=3)
            #---------------------------------------------------------------------------------#

          }
